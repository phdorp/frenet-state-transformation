41 Points<NumQueries> posFrenet(const Points<NumQueries> &posCartes) const {
42 // lengths from path origin to Cartesian positions
43 const auto lengths{m_path->lengths(posCartes)};
44 // next points on path to Cartesian positions

46 // vectors from query points to path points

182 velTransform(const Points<NumQueries> &posFrenet) const {
183 const auto tangents{m_path->tangent(posFrenet.x())};
184 const auto normals{m_path->normal(posFrenet.x())};
185 const auto curvs{m_path->angle1(posFrenet.x())};

187 {tangents.y() * (1 - curvs * posFrenet.y()), normals.y()}};

200 accTransform(const Points<NumQueries> &velFrenet,
201 const Points<NumQueries> &posFrenet) const {
202 const auto tangents{m_path->tangent(posFrenet.x())};
203 const auto normals{m_path->normal(posFrenet.x())};
204 // scaling factor due to lateral position from path and path curvature
205 const auto curvs{m_path->angle1(posFrenet.x())};
206 const auto latScale{1 - curvs * posFrenet.y()};
207 // scaling factor derivatvie
208 const auto curv1s{m_path->angle2(posFrenet.x())};
209 const auto latScaleDer{curv1s * velFrenet.x() * posFrenet.y() +
210 curvs * velFrenet.y()};
211 return {{normals.x() * curvs * latScale * velFrenet.x() -
212 tangents.x() * latScaleDer,
213 -curvs * tangents.x() * velFrenet.x()},
214 {normals.y() * curvs * latScale * velFrenet.x() -
215 tangents.y() * latScaleDer,
216 -curvs * tangents.y() * velFrenet.x()}};
217 }
