56 Points<NumQueries> operator()(const ArrayQueries &lengths) const override {
57 // indices of corresponding polychain segments
58 Eigen::Array<int, NumQueries, 1> indicesLengths{indices(lengths)};
59 for (int &idx : indicesLengths)
60 idx = idx > 0 ? idx - 1 : idx;
61
62 // relative position along the linear segment
63 const ArrayPoints pathLengthsdiff{diffBackward(m_lengths)};
64 ArrayQueries segmentPart{(lengths - m_lengths(indicesLengths)) /
65 pathLengthsdiff(indicesLengths + 1)};
66 for (double &part : segmentPart)
67 part = std::clamp(part, 0.0, 1.0);
68
69 // absolute position along path
70 ArrayQueries x{m_points[0].x()(indicesLengths + 1) * segmentPart +
71 m_points[0].x()(indicesLengths) * (1 - segmentPart)};
72 ArrayQueries y{m_points[0].y()(indicesLengths + 1) * segmentPart +
73 m_points[0].y()(indicesLengths) * (1 - segmentPart)};
74
75 return {x, y};

86 ArrayQueries lengths(const Points<NumQueries> &points) const override {
87 ArrayQueries lengthsPoints(points.numPoints());
88
89 // determine lengths for all query points
90 for (int cQuery{}; cQuery < points.numPoints(); ++cQuery) {
91 // current minimum squared distance between query point and polychain
92 double distanceSquare{-1.0};
93
94 // determine distance between query point and path segment
95 for (int cPoints{1}; cPoints < m_numPoints; ++cPoints) {
96 // point at end of current segment
97 const Point nextPoint{m_points[0](cPoints)};
98 // difference between "nextPoint" and query point
99 const Point diffPoint{nextPoint - points(cQuery)};
100 // parameter determining next point along current linear segment
101 const double segmentPart{(diffPoint.x() * m_xDiff(cPoints) +
102 diffPoint.y() * m_yDiff(cPoints)) /
103 m_diffSquare(cPoints)};
104
105 // squared distance between current segment and query point
106 double distanceSquareCand{};
107 // length along polychain of shortest distance point on segment to query
108 // point
109 double lengthCand{};
110
111 // determine squared distance and length if shortest distance point is
112 // beyond segment end
113 if (segmentPart >= 1.0) {
114 distanceSquareCand =
115 m_points[0](cPoints - 1).distanceSquare(points(cQuery));
116 lengthCand = m_lengths.data()[cPoints - 1];
117 }
118 // determine squared distance and length if shortest distance point is
119 // beyond segment start
120 else if (segmentPart <= 0.0) {
121 distanceSquareCand =
122 m_points[0](cPoints).distanceSquare(points(cQuery));
123 lengthCand = m_lengths.data()[cPoints];
124 }
125 // determine squared distance and length if shortest distance point is
126 // on segment
127 else {
128 // point at start of current segment
129 const Point prevPoint{m_points[0](cPoints - 1)};
130 // shortest distance point on current segment
131 const Point pointCand{
132 prevPoint.x() * segmentPart + (1 - segmentPart) * nextPoint.x(),
133 prevPoint.y() * segmentPart + (1 - segmentPart) * nextPoint.y()};
134 distanceSquareCand = points(cQuery).distanceSquare(pointCand);
135 lengthCand = m_lengths(cPoints - 1) + pointCand.distance(prevPoint);
136 }
137
138 // update length and squared distance if squared distance is smaller
139 // than current minimum
140 if (distanceSquareCand < distanceSquare || distanceSquare < 0) {
141 distanceSquare = distanceSquareCand;
142 lengthsPoints(cQuery) = lengthCand;
143 }
144 }
145 }
146
147 return lengthsPoints;
148 }
